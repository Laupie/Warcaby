class Warcaby:
    def __init__(self):
        self.rozmiar = 8
        self.plansza = self.stworz_plansze()

    def stworz_plansze(self):
        plansza = [[' ' for _ in range(self.rozmiar)] for _ in range(self.rozmiar)]
        for wiersz in range(self.rozmiar):
            for kolumna in range(self.rozmiar):
                if (wiersz + kolumna) % 2 == 1:  # Tylko na czarnych polach
                    if wiersz < 3:
                        plansza[wiersz][kolumna] = 'C'  # Pionki gracza 1 (Czerwony)
                    elif wiersz > 4:
                        plansza[wiersz][kolumna] = 'B'  # Pionki gracza 2 (Biały)
        return plansza

    def wyswietl_plansze(self):
        for wiersz in self.plansza:
            print(' '.join(wiersz))
        print('\n')

    def ruch(self, gracz, start, koniec):
        x1, y1 = start
        x2, y2 = koniec

        if self.plansza[x1][y1].upper() != gracz:
            print("Nieprawidłowy pionek!")
            return False

        if self.plansza[x2][y2] != ' ':
            print("Pole docelowe zajęte!")
            return False

        # Sprawdzenie poprawności ruchu
        dx, dy = x2 - x1, y2 - y1
        if abs(dx) == 1 and abs(dy) == 1:
            # Zwykły ruch
            self.plansza[x2][y2] = self.plansza[x1][y1]
            self.plansza[x1][y1] = ' '
            return True
        elif abs(dx) == 2 and abs(dy) == 2:
            # Zbijanie
            srodek_x, srodek_y = (x1 + x2) // 2, (y1 + y2) // 2
            if self.plansza[srodek_x][srodek_y].upper() != gracz and self.plansza[srodek_x][srodek_y] != ' ':
                self.plansza[x2][y2] = self.plansza[x1][y1]
                self.plansza[x1][y1] = ' '
                self.plansza[srodek_x][srodek_y] = ' '
                return True

        print("Nieprawidłowy ruch!")
        return False

    def sprawdz_zwyciezce(self):
        pionki_c = sum(wiersz.count('C') for wiersz in self.plansza)
        pionki_b = sum(wiersz.count('B') for wiersz in self.plansza)
        if pionki_c == 0:
            return "Biały wygrywa!"
        elif pionki_b == 0:
            return "Czerwony wygrywa!"
        return None

    def graj(self):
        print("Witaj w grze w warcaby!")
        gracz = 'C'  # Czerwony zaczyna
        while True:
            self.wyswietl_plansze()
            zwyciezca = self.sprawdz_zwyciezce()
            if zwyciezca:
                print(zwyciezca)
                break

            print(f"Tura gracza: {'Czerwony' if gracz == 'C' else 'Biały'}")
            try:
                start = tuple(map(int, input("Podaj pole startowe (wiersz, kolumna): ").split(',')))
                koniec = tuple(map(int, input("Podaj pole końcowe (wiersz, kolumna): ").split(',')))
                if self.ruch(gracz, start, koniec):
                    gracz = 'B' if gracz == 'C' else 'C'  # Zmiana gracza
            except Exception as e:
                print("Nieprawidłowe dane wejściowe. Spróbuj ponownie!")

warcaby = Warcaby()
warcaby.graj()
